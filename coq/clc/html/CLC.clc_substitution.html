Library CLC.clc_substitution

From mathcomp Require Import ssreflect ssrbool eqtype ssrnat seq.
From Coq Require Import ssrfun Program Utf8 Classical.
Require Import AutosubstSsr ARS
  clc_context clc_ast clc_confluence clc_subtype clc_typing
  clc_weakening.

Set Implicit Arguments.

Reserved Notation "[ Δ |- σ -| Γ ]".

Inductive agree_subst :
  context term → (var → term) → context term → Prop :=
| agree_subst_nil σ :
  [ nil |- σ -| nil ]
| agree_subst_u Δ σ Γ A :
  [ Δ |- σ -| Γ ] →
  [ A.[σ] +u Δ |- up σ -| A +u Γ ]
| agree_subst_l Δ σ Γ A :
  [ Δ |- σ -| Γ ] →
  [ A.[σ] +l Δ |- up σ -| A +l Γ ]
| agree_subst_n Δ σ Γ :
  [ Δ |- σ -| Γ ] →
  [ □ Δ |- up σ -| □ Γ ]
| agree_subst_wkU Δ σ Γ n A :
  [ Δ |- σ -| Γ ] →
  [ %Δ |- n :- A.[σ] ] →
  [ Δ |- n .: σ -| A +u Γ ]
| agree_subst_wkL Δ1 Δ2 Δ σ Γ n A :
  merge Δ1 Δ2 Δ →
  [ Δ1 |- σ -| Γ ] →
  [ Δ2 |- n :- A.[σ] ] →
  [ Δ |- n .: σ -| A +l Γ ]
| agree_subst_wkN Δ σ Γ n :
  [ Δ |- σ -| Γ ] →
  [ Δ |- n .: σ -| □ Γ ]
| agree_subst_convU Δ σ Γ A B l :
  A <: B →
  [ %Δ |- B.[ren (+1)].[σ] :- Sort U l ] →
  [ Δ |- σ -| A +u Γ ] →
  [ Δ |- σ -| B +u Γ ]
| agree_subst_convL Δ σ Γ A B l :
  A <: B →
  [ %Δ |- B.[ren (+1)].[σ] :- Sort L l ] →
  [ %Γ |- B :- Sort L l ] →
  [ Δ |- σ -| A +l Γ ] →
  [ Δ |- σ -| B +l Γ ]
where "[ Δ |- σ -| Γ ]" := (agree_subst Δ σ Γ).

Lemma agree_subst_pure Δ σ Γ :
  [ Δ |- σ -| Γ ] → pure Γ → pure Δ.

Lemma agree_subst_refl Γ :
  [ Γ |- ids -| Γ ].

Lemma agree_subst_hasU Δ σ Γ :
  [ Δ |- σ -| Γ ] →
  ∀ x A,
    [ x :u A ∈ Γ ] →
    [ Δ |- σ x :- A.[σ] ].

Lemma agree_subst_hasL Δ σ Γ :
  [ Δ |- σ -| Γ ] →
  ∀ x A,
    [ x :l A ∈ Γ ] →
    [ Δ |- σ x :- A.[σ] ].

Lemma agree_subst_re_re Δ σ Γ :
  [ Δ |- σ -| Γ ] →
  [ %Δ |- σ -| %Γ ].

Lemma merge_agree_subst_inv Δ σ Γ :
  [ Δ |- σ -| Γ ] →
  ∀ Γ1 Γ2,
    [ Γ1 ‡ Γ2 ‡ Γ ] →
    ∃ Δ1 Δ2,
      [ Δ1 ‡ Δ2 ‡ Δ ] ∧ [ Δ1 |- σ -| Γ1 ] ∧ [ Δ2 |- σ -| Γ2 ].

Lemma substitution Γ m A :
  [ Γ |- m :- A ] →
  ∀ Δ σ,
    [ Δ |- σ -| Γ ] →
    [ Δ |- m.[σ] :- A.[σ] ].

Lemma substitutionU Γ1 m A B :
  [ A +u Γ1 |- m :- B ] →
  ∀ Γ2 Γ n,
    [ Γ2 ] →
    [ Γ1 ‡ Γ2 ‡ Γ ] →
    [ Γ2 |- n :- A ] →
    [ Γ |- m.[n/] :- B.[n/] ].

Lemma substitutionN Γ1 m A B :
  [ □ Γ1 |- m :- B ] →
  ∀ Γ2 n,
    [ Γ2 |- n :- A ] →
    [ Γ1 |- m.[n/] :- B.[n/] ].

Lemma substitutionL Γ1 m A B :
  [ A +l Γ1 |- m :- B ] →
  ∀ Γ2 Γ n,
    [ Γ1 ‡ Γ2 ‡ Γ ] →
    [ Γ2 |- n :- A ] →
    [ Γ |- m.[n/] :- B.[n/] ].

Lemma context_convU Γ m A B C l :
  B === A →
  [ %Γ |- A :- Sort U l ] →
  [ A +u Γ |- m :- C ] →
  [ B +u Γ |- m :- C ].

Lemma context_convL Γ m A B C l :
  B === A →
  [ %Γ |- A :- Sort L l ] →
  [ A +l Γ |- m :- C ] →
  [ B +l Γ |- m :- C ].
