Set Implicit Arguments.

Reserved Notation "[ x ∈ Γ ]".
Inductive isL : context term → nat → Prop :=
| isL_O Γ A :
  [ 0 ∈ A +l Γ ]
| isL_S Γ i A s :
  [ i ∈ Γ ] →
  [ i.+1 ∈ A +{s} Γ ]
| isL_N Γ i :
  [ i ∈ Γ ] →
  [ i.+1 ∈ □ Γ ]
where "[ x ∈ Γ ]" := (isL Γ x).

Reserved Notation "[ x ∉ Γ ]".
Inductive isN : context term → nat → Prop :=
| isN_O Γ :
  [ 0 ∉ □ Γ ]
| isN_S Γ i A s :
  [ i ∉ Γ ] →
  [ i.+1 ∉ A +{s} Γ ]
| isN_N Γ i :
  [ i ∉ Γ ] →
  [ i.+1 ∉ □ Γ ]
where "[ x ∉ Γ ]" := (isN Γ x).

Fixpoint occurs (i : nat) (m : term) : nat :=
  match m with
  | Var x ⇒ if x == i then 1 else 0
  | Sort _ _ ⇒ 0
  | Prod A B _ ⇒ occurs i A + occurs (i.+1) B
  | Lolli A B _ ⇒ occurs i A + occurs (i.+1) B
  | Lam m ⇒ occurs (i.+1) m
  | App m n ⇒ occurs i m + occurs i n
  end.

Lemma eqn_refl (n : nat) : n == n.

Lemma isL_pure Γ i : [ i ∈ Γ ] → ¬[ Γ ].

Lemma isL_hasU Γ i :
  [ i ∈ Γ ] → ∀ x A, ¬[ x :u A ∈ Γ ].

Lemma isL_hasL Γ i :
  [ i ∈ Γ ] → ∀ x A, [ x :l A ∈ Γ ] → x = i.

Lemma isN_hasU Γ i :
  [ i ∉ Γ ] → ∀ x A, [ x :u A ∈ Γ ] → x == i = false.

Lemma isN_hasL Γ i :
  [ i ∉ Γ ] → ∀ x A, [ x :l A ∈ Γ ] → x == i = false.

Lemma isL_merge_inv Γ1 Γ2 Γ :
  [ Γ1 ‡ Γ2 ‡ Γ ] →
    ∀ i, [ i ∈ Γ ] →
      ([ i ∈ Γ1 ] ∧ [ i ∉ Γ2 ]) ∨
      ([ i ∈ Γ2 ] ∧ [ i ∉ Γ1 ]).

Lemma isN_merge_inv Γ1 Γ2 Γ :
  [ Γ1 ‡ Γ2 ‡ Γ ] →
    ∀ i, [ i ∉ Γ ] →
      [ i ∉ Γ1 ] ∧ [ i ∉ Γ2 ].

Lemma narity Γ m A :
  [ Γ |- m :- A ] →
    ∀ i, [ i ∉ Γ ] → occurs i m = 0.

Theorem linearity Γ m A :
  [ Γ |- m :- A ] →
    ∀ i, [ i ∈ Γ ] → occurs i m = 1.

Theorem promotion Γ m A B s :
  [ Γ ] →
  [ Γ |- ] →
  [ Γ |- m :- Lolli A B s ] →
  ∃ m, [ Γ |- m :- Prod A B s ].

Theorem dereliction Γ m A B s :
  [ Γ |- ] →
  [ Γ |- m :- Prod A B s ] →
  ∃ m, [ Γ |- m :- Lolli A B s ].
