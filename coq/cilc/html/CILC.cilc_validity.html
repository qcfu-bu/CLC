Lemma merge_context_ok_inv Γ1 Γ2 Γ :
  merge Γ1 Γ2 Γ →
  [ Γ |- ] →
  [ Γ1 |- ] ∧ [ Γ2 |- ].

Theorem validity Γ m A :
  [ Γ |- ] →
  [ Γ |- m :- A ] →
  ∃ s l, [ re Γ |- A :- Sort s l ].
