Ltac solve_sub :=
  match goal with
  | [ H : _ <: _ |- _ ] ⇒
    let A := fresh "A" in
    let B := fresh "B" in
    let sb := fresh "sb" in
    let c1 := fresh "c1" in
    let c2 := fresh "c2" in
    destruct H as [A B sb c1 c2]; destruct sb
  end;
  match goal with
  | [ c1 : ?A === ?x, c2 : ?x === ?B |- _ ] ⇒
    assert (A === B) by
      (eapply conv_trans; try solve [apply c1| apply c2]);
    clear c1 c2;
    solve_conv
  | _ ⇒ solve_conv
  end.

Lemma u_Sort_inv Γ s l A :
  [ Γ |- s @ l :- A ] → U @ l.+1 <: A.

Lemma u_Var_inv Γ x B :
  [ Γ |- Var x :- B ] →
  (∃ A, hasU Γ x A ∧ A <: B) ∨
  (∃ A, hasL Γ x A ∧ A <: B).

Lemma l_Var_inv Γ A B :
  [ A +l Γ |- Var 0 :- B ] → A.[ren (+1)] <: B.

Lemma u_Prod_inv Γ A B C :
  [ Γ |- Prod A B U :- C ] →
  ∃ s l l',
    [ Γ |- A :- Sort U l ] ∧
    [ A +u Γ |- B :- Sort s l ] ∧
    Sort U l' <: C.

Lemma l_Prod_inv Γ A B C :
  [ Γ |- Prod A B L :- C ] →
  ∃ s l l',
    [ Γ |- A :- Sort L l ] ∧
    [ +n Γ |- B :- Sort s l ] ∧
    Sort U l' <: C.

Lemma u_Lolli_inv Γ A B C :
  [ Γ |- Lolli A B U :- C ] →
  ∃ s l l',
    [ Γ |- A :- Sort U l ] ∧
    [ A +u Γ |- B :- Sort s l ] ∧
    Sort L l' <: C.

Lemma l_Lolli_inv Γ A B C :

Lemma u_Lam_invX Γ A0 A1 B C s0 s1 m t l :
  [ Γ |- Lam A0 m s0 :- C ] →
  (C <: Prod A1 B s1 ∧ [ re (A1 +{s1} Γ) |- B :- Sort t l ]) →
  [ A1 +{s1} Γ |- m :- B ].

Lemma s_Ind_invX Γ A B Cs s :
  [ Γ |- Ind A Cs s :- B ] →
  ∃ l,
    A <: B ∧
    arity s A ∧
    List.Forall (constr 0 s) Cs ∧
    pure Γ ∧
    [ Γ |- A :- Sort U l ] ∧
    List.Forall (fun C ⇒ [ A +u Γ |- C :- Sort U l ]) Cs.

Lemma s_Ind_inv Γ A Cs s :
  [ Γ |- Ind A Cs s :- A ] →
  ∃ l,
    arity s A ∧
    List.Forall (constr 0 s) Cs ∧
    pure Γ ∧
    [ Γ |- A :- Sort U l ] ∧
    List.Forall (fun C ⇒ [ A +u Γ |- C :- Sort U l ]) Cs.

Lemma s_Constr_invX Γ i I CI :
  [ Γ |- Constr i I :- CI ] →
  ∃ A C Cs s,
    iget i Cs C ∧
    pure Γ ∧
    I = Ind A Cs s ∧
    C.[I/] <: CI ∧
    [ Γ |- I :- A ].
